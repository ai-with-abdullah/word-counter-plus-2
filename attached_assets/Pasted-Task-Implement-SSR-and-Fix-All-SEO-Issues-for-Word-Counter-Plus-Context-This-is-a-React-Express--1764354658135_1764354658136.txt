Phase 2: Implement Server-Side Rendering (SSR)
This is the MAIN fix that resolves 90% of issues.

Create these new files:

client/src/entry-server.tsx - Server-side render entry
import { renderToString } from 'react-dom/server';
import { Router } from 'wouter';
import staticLocationHook from 'wouter/static-location';
import App from './App';
export function render(url: string) {
  const html = renderToString(
    <Router hook={staticLocationHook(url)}>
      <App />
    </Router>
  );
  return { html };
}

client/src/entry-client.tsx - Client hydration entry
